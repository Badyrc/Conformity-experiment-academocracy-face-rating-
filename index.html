// CLEAN VERSION STAGE 1 const jsPsych = initJsPsych(); function downloadText(filename, text) { const blob = new Blob([text], { type: 'text/csv' }); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = filename; a.click(); } jsPsych.run([ { type: jsPsychCallFunction, func: () => { const raw = jsPsych.data.get().csv(); downloadText("stage1_raw.csv", raw); } } ]);
